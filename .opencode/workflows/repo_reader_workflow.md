---
name: repo_reader_workflow
type: workflow
description: |
  Complete workflow for reading and documenting a code repository.
  Orchestrates the Broad Summary Agent, Connection Builder Agent, and Snippet Builder Agent
  to generate comprehensive documentation including independent summaries, dependency graphs,
  and deep-dive documentation with highlighted code snippets.

version: "1.0.0"

inputs:
  - name: repository_path
    type: string
    description: Absolute path to the target repository to analyze
    required: true
  - name: output_base_directory
    type: string
    description: Base directory where all outputs will be stored
    required: true
    default: ./output

steps:
  - name: step_1_parallel_analysis
    description: Execute BSA and CBA simultaneously to analyze the codebase
    parallel: true
    steps:
      - name: run_broad_summary_agent
        agent: broad_summary_agent
        inputs:
          repository_path: "${inputs.repository_path}"
          output_directory: "${inputs.output_base_directory}/summaries"
        outputs:
          summary_files: "${outputs.summary_files}"
      
      - name: run_connection_builder_agent
        agent: connection_builder_agent
        inputs:
          repository_path: "${inputs.repository_path}"
          output_directory: "${inputs.output_base_directory}/mermaid_graphs"
        outputs:
          mermaid_files: "${outputs.mermaid_files}"
  
  - name: step_2_snippet_builder
    description: Execute SB to generate final deep-dive documentation
    depends_on:
      - step_1_parallel_analysis
    agent: snippet_builder_agent
    inputs:
      repository_path: "${inputs.repository_path}"
      summary_files_directory: "${inputs.output_base_directory}/summaries"
      mermaid_files_directory: "${inputs.output_base_directory}/mermaid_graphs"
      output_directory: "${inputs.output_base_directory}/deep_dives"
    outputs:
      documentation_files: "${outputs.documentation_files}"

outputs:
  - name: summary_files
    type: array
    description: Paths to all summary files generated by BSA
    items:
      type: string
  
  - name: mermaid_files
    type: array
    description: Paths to all Mermaid.js graph files generated by CBA
    items:
      type: string
  
  - name: documentation_files
    type: array
    description: Paths to all deep-dive documentation files generated by SB
    items:
      type: string
  
  - name: output_directories
    type: object
    description: Map of output directories for all artifacts
    properties:
      summaries:
        type: string
        description: Directory containing concept summaries
      mermaid_graphs:
        type: string
        description: Directory containing dependency graphs
      deep_dives:
        type: string
        description: Directory containing deep-dive documentation
---
