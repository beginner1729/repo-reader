---
name: repo_reader_workflow
type: workflow
description: |
  Complete workflow for reading and documenting a code repository.
  Orchestrates graphify (knowledge graph builder), Broad Summary Agent, and Snippet Builder Agent
  to generate comprehensive documentation including independent summaries, knowledge graphs,
  and deep-dive documentation with highlighted code snippets.

version: "2.0.0"

inputs:
  - name: repository_path
    type: string
    description: Absolute path to the target repository to analyze
    required: true
  - name: output_base_directory
    type: string
    description: Base directory where all outputs will be stored
    required: true
    default: ./output
  - name: graphify_output_directory
    type: string
    description: Directory where graphify outputs will be stored
    required: true
    default: ./graphify-out

steps:
  - name: step_1_graphify
    description: Run graphify first to build the knowledge graph
    command: graphify
    inputs:
      path: "${inputs.repository_path}"
      output: "${inputs.graphify_output_directory}"
      flags: "--directed"
    outputs:
      graph_json: "${inputs.graphify_output_directory}/graph.json"
      graph_report: "${inputs.graphify_output_directory}/GRAPH_REPORT.md"

  - name: step_2_broad_summary
    description: Execute BSA to analyze and summarize concepts
    depends_on:
      - step_1_graphify
    agent: broad_summary_agent
    inputs:
      repository_path: "${inputs.repository_path}"
      output_directory: "${inputs.output_base_directory}/summaries"
    outputs:
      summary_files: "${outputs.summary_files}"

  - name: step_3_snippet_builder
    description: Execute SB to generate final deep-dive documentation
    depends_on:
      - step_2_broad_summary
    agent: snippet_builder_agent
    inputs:
      repository_path: "${inputs.repository_path}"
      summary_files_directory: "${inputs.output_base_directory}/summaries"
      graphify_files_directory: "${inputs.graphify_output_directory}"
      output_directory: "${inputs.output_base_directory}/deep_dives"
    outputs:
      documentation_files: "${outputs.documentation_files}"

outputs:
  - name: summary_files
    type: array
    description: Paths to all summary files generated by BSA
    items:
      type: string

  - name: graphify_files
    type: array
    description: Paths to graphify output files (graph.json, GRAPH_REPORT.md, graph.html)
    items:
      type: string

  - name: documentation_files
    type: array
    description: Paths to all deep-dive documentation files generated by SB
    items:
      type: string

  - name: output_directories
    type: object
    description: Map of output directories for all artifacts
    properties:
      summaries:
        type: string
        description: Directory containing concept summaries
      graphify_output:
        type: string
        description: Directory containing knowledge graph outputs
      deep_dives:
        type: string
        description: Directory containing deep-dive documentation
---
